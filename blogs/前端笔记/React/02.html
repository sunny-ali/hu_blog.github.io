react hooks中主要的API使用

银河系小仙女 2020-06-08 React

什么是Hooks? hooks 是react 16.8 新增的特性，它允许你在不写class的情况下操作state 和react的其他特性。 hooks 只是多了一种写组件的方法，使编写一个组件更简单更方便，同时可以自定义hook把公共的逻辑提取出来，让逻辑在多个组件之间共享。

# 接下来主要通过小demo来学习 Hooks：

# useState：状态

示例如下：

 //通过useState Hook 声明了一个新的state变量
  const [count,setCount] = useState(0)
  //这种js语法叫做数组解构，意味着我们同时创建了count和setCount两个变量
  // count的值为useState返回的第一个值，setCount是返回的第二个值
  //相当于以下代码：
  // var countStateVariable = useState(0); // 返回一个有两个元素的数组
  // var count = countStateVariable[0]; // 数组里的第一个值
  // var setcount = countStateVariable[1]; // 数组里的第二个值

  const [obj,setObj] = useState({name:'Ali'})
  const [arr,setArr] = useState([1,2,3])
  const [func,setFunc] = useState(()=>{return {name:'ali'};})

  return (
    <div>
      <p>{count}</p>
      {/*当用户点击按钮后，传递一个新的值给setCount，react会重新渲染App组件把最新的值传给它*/}
      <button onClick={()=>setCount(count+1)}>点我</button>

      <p>{obj.name}--{obj.age}</p>
      <button onClick={()=>setObj({name:'mini',age:17})}>点我</button>

      <p>{arr}</p>
      <button onClick={()=>setArr([2,3,4])}>点我</button>

      <p>{func.name}</p>
      {/*setFunc返回的也必须是同类型*/}
      <button onClick={()=>setFunc(()=>{return {name:'yli'}})}>点我</button>
    </div>
  );
}

export default FunComp;

# useEffect：副作用

以下特点： 1，看做 componentDidMount，componentDidUpdate 和 componentWillUnmount 这三个函数的组合 2，副作用（DOM操作，数据请求，组件更新） 3，采用闭包形式，在组件函数内部执行，因此可以获取props和state 4，无阻塞更新 5，useEffect(回调函数，数组(可不写)) 6，多个useEffect

例子：点击按钮自加一，且在count能被2整除时控制台输出当前的count 第一种写法：

import React from 'react'

class  ClassComp extends React.Component{
  state = {
    count:0
  }
  componentDidMount() {//插入真实dom完成初始化渲染
    if(this.state.count%2===0){
      console.log(this.state.count)
    }
  }
  componentDidUpdate() {//完成更新
    if(this.state.count%2===0){
      console.log(this.state.count)
    }
  }
  render() {
    const {count} = this.state
    return(
      <div>
        <p>{count}</p>
        <button onClick={()=>this.setState({count:count+1})}>点击加一</button>
      </div>
    )
  }
}
export default ClassComp;

第二种写法：使用useEffect

import React,{useState,useEffect} from 'react'

export default function FunComp() {
  const [count, setCount] = useState(0);

  //useEffect有二个参数：函数，数组[]代表的是能够引起副作用执行的依赖
  useEffect(()=>{
    if(count%2===0){
      console.log(count)
    }
  },
    //如果不写第二个参数则默认只有通过useState Hook 声明了一个新的state变量发生变化都会调用此副作用函数
    // []//只在组件中挂载时执行一次
    [count]//每次count发生改变都会调用此副作用
  )
  return(
      <div>
        <p>{count}</p>
        <button onClick={()=>setCount(count+1)}>点击加一</button>
      </div>
  )
}

# useReducer：复杂的副作用处理

import React,{useReducer} from 'react'

export default function ReducerDemo() {
  const initialState = {count: 0};
  const reducer = (state,action) =>{
    switch (action.type) {
      case 'increment':
        return {count: state.count + action.payload};
      case 'decrement':
        return {count: state.count - 1};
      default:
        throw new Error();
    }
  }
  const [state, dispatch] = useReducer(reducer, initialState);
  return(
    <div>
      reducer {state.count}                                             
                                             {/*{自加2}*/}
      <button onClick={() => dispatch({type: 'increment',payload:2})}>+</button>
      <button onClick={() => dispatch({type: 'decrement'})}>-</button>
    </div>
  )
}

# useContext：深层状态传递

例子：父级给子级的子级传值 第一种：通过最原始方法props逐层传递：

import React,{useState} from 'react'

const Child = (props) => {
  const style = {
    backgroundColor:props.bgColor
  }
  return <div style={style}>Child</div>
}

const Demo = (props) => {
  const {bgColor} = props
  return <Child bgColor={bgColor}/>
}

export default function UseContextDemo() {
  const [bgColor,setBgColor] =useState('black')
  return(
    <div>
      <button onClick={()=>setBgColor('pink')}>换背景色</button>
      <Demo bgColor={bgColor}/>
    </div>
  )
}

第二种:使用UseContext：

import React,{useState,useContext} from 'react'

const themContext = React.createContext('black')

const Child = () => {
  const bgColor = useContext(themContext)
  const style = {
    backgroundColor:bgColor
  }
  return <div style={style}>Child</div>
}

const Demo = () => {
  return <Child/>
}

export default function UseContextDemo() {
  const [bgColor,setBgColor] =useState('black')
  return(
    <div>
      <button onClick={()=>setBgColor('pink')}>换背景色</button>
      <themContext.Provider value={bgColor}>
        <Demo/>
      </themContext.Provider>
    </div>
  )
}

# 自定义Hooks

通过自定义 Hook，可以将组件逻辑提取到可重用的函数中 自定义hooks-useCount代码：

import React, {useEffect, useState} from "react";

export  default function useCount(){//自定义hook
  const [count,setCount] = useState(0);
  useEffect(()=>{
    setInterval(()=>{
      setCount(count => count + 1)
    })
  },[])
  console.log(count)
  return count;
}

# useRef：引用

例子：刷新输入框自动聚焦

import React, {useEffect, useRef} from "react";

function Input() {
  const inputRef = useRef()

  useEffect(()=>{
    inputRef.current.focus()
  },[])

  return <input ref={inputRef}/>
}
export default function RefDemo() {
  return(
    <div>
      <Input/>
    </div>
  )
}

# useCallback&useMeno：

useCallback：一般使用在存储函数，它存储于函数体，当你的依赖不发生更改时，函数体也不会发生更改。 useMeno：根据依赖来计算。

import React, {useState, useEffect, useCallback, useMemo} from 'react'

export default function CallbackMemo() {
  const [count,setCount] =useState(0)
  const [otherCount,setOtherCount] =useState(0)
  useEffect(()=>{
    setInterval(()=>{
      setOtherCount((otherCount)=> otherCount + 1);
    },30)
    setInterval(()=>{
      setCount(count => count + 1)
    },1000)
  },[])//只执行一次
  console.log('更新...')

  //依赖不发生更改时，函数体也不会发生更改
  // useCallback 将函数缓存起来 节流
  const handleClick = useCallback((ev) =>{
    console.log(ev.target)
  },[])

  //const result = count * 100//当setCount和setOtherCount发生改变时都会执行这行代码，

  //当其它状态更改时会触发了组件的更新，
  // 但是我的这个值只依赖于某一个属性某一个状态时用useMemo
  const result = useMemo(()=>{
    console.log('计算中.......')
    return count * 100
  },[count])//只依赖于count

  return(
    <div onClick={handleClick}>
      {count} {otherCount} {result}
    </div>
  )
}
